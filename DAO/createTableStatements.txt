CREATE TABLE Users
(
    UserName varchar(255) not null primary key,
    Password varchar(255),
    Email varchar(255),
    FirstName varchar(255),
    LastName varchar(255),
    Gender varchar(255),
    PersonID varchar(255) not null foreign key,
);

CREATE TABLE Persons
(
    PersonID varchar(255) not null primary key,
    Descendant varchar(255) not null foreign key,
    FirstName varchar(255),
    LastName varchar(255),
    Gender varchar(255),
    FatherID varchar(255),
    MotherID varchar(255),
    SpouseID varchar(255),
);

CREATE TABLE Events
(
    EventID varchar(255) not null primary key,
    Descendant varchar(255),
    PersonID varchar(255) not null foreign key,
    Latitude varchar(255),
    Longitude varchar(255),
    Country varchar(255),
    City varchar(255),
    EventType varchar(255),
    Year varchar(255),
);

CREATE TABLE AuthTokens
(
    AuthID varchar(255) not null primary key,
    UserName varchar(255) not null foreign key,
)

create table members
(
	id integer not null primary key autoincrement,
	name varchar(255) not null,
	email_address varchar(255) not null
);

create table book
(
	id integer not null primary key autoincrement,
	title varchar(255) not null,
	author varchar(255) not null,
	genre varchar(32) not null,
	constraint ck_genre check (genre in ('Unspecified', 'Fiction', 'NonFiction', 'HistoricalFiction'))
);


create table reading
(
	member_id int not null,
	book_id int not null
);


create table categories
(
	id integer not null primary key autoincrement,
	name varchar(255) not null,
	parent_id int
);


create table category_book
(
	category_id int not null,
	book_id int not null
);
